institution.routes.js - Geospatial Query
/* Find properties near an institution */ SELECT p.property_id, p.title, ROUND(CAST(ST_Distance(i.campus_location::geography, p.location::geography) / 1000 AS numeric), 2) AS distance_km FROM institutions i JOIN properties p ON ST_DWithin(i.campus_location::geography, p.location::geography, $2 * 1000) -- radius in meters WHERE i.institution_id = $1 AND p.status = 'ACTIVE' ORDER BY distance_km ASC;
